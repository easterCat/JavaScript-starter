    function ListNode(val) {
      this.val = val;
      this.next = null;
    }

    let l1 = {
      val: 2,
      next: {
        val: 4,
        next: {
          val: 3,
          next: null
        }
      }
    };

    let l2 = {
      val: 5,
      next: {
        val: 6,
        next: {
          val: 4,
          next: null
        }
      }
    };

    console.log("result", addTwoNumbers(l1, l2));

    function addTwoNumbers(l1, l2) {
      let prev = l1;
      let next = l2;
      let curr = new ListNode(0);
      let center = curr;
      let carry = 0;

      while (prev || next) {
        let x = 0;
        let y = 0;
        if (prev) {
          x = prev.val;
        }
        if (next) {
          y = next.val;
        }
        let total = carry + x + y;
        if (total > 9) {
          center.next = new ListNode(total % 10);
          carry = 1;
        } else {
          center.next = new ListNode(total);
          carry = 0;
        }

        center = center.next;

        if (prev) {
          prev = prev.next;
        }
        if (next) {
          next = next.next;
        }
      }

      if (carry > 0) {
        center.next = new ListNode(1);
      }

      return curr.next;
    }
  </script>
